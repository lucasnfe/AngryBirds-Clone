664  float levelLeftBound = _groundTransform.transform.position.x - _groundTransform.GetComponent<Collider2D>().bounds.size.x/2f;
665  float groundSurfacePos = _groundTransform.transform.position.x + _groundTransform.GetComponent<Collider2D>().bounds.size.y/2f;

669  float maxPosY = _currentLevel.gameObjects[0].Position.y + _currentLevel.gameObjects[0].GetBounds().size.y/2f;
670 
671  // Get position of first non-empty stack
672  for(int i = 0; i < _currentLevel.gameObjects.Count; i++)
673  {
674  float minPosXCandidate = _currentLevel.gameObjects[i].Position.x - _currentLevel.gameObjects[i].GetBounds().size.x/2f;
675  if(minPosXCandidate < minPosX)
676  minPosX = minPosXCandidate;
677 
678  float maxPosXCandidate = _currentLevel.gameObjects[i].Position.x + _currentLevel.gameObjects[i].GetBounds().size.x/2f;
679  if(maxPosXCandidate > maxPosX)
680  maxPosX = maxPosXCandidate;
681 
682  float maxPosYCandidate = _currentLevel.gameObjects[i].Position.y + _currentLevel.gameObjects[i].GetBounds().size.y/2f;
683  if(maxPosYCandidate > maxPosY)
684  maxPosY = maxPosYCandidate;
685  }
686 
687  float cameraWidth = Mathf.Abs(minPosX - levelLeftBound) +
688  Mathf.Max(Mathf.Abs(maxPosX - minPosX), Mathf.Abs(maxPosY - groundSurfacePos)) + 0.5f;
689 
690  _camera.SetCameraWidth(cameraWidth);
